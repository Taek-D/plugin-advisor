     ```
     const searchParams = useSearchParams();
     const router = useRouter();
     const rawType = searchParams.get("type");
     const initialType: ActiveType = rawType === "mcp" || rawType === "plugin" ? rawType : "all";
     ```
   - Add state: `const [activeType, setActiveType] = useState<ActiveType>(initialType);`
   - Add handler (using useCallback with [searchParams, router] deps):
     ```
     const handleTypeChange = useCallback((type: ActiveType) => {
       setActiveType(type);
       const params = new URLSearchParams(searchParams.toString());
       if (type === "all") {
         params.delete("type");
       } else {
         params.set("type", type);
       }
       const qs = params.toString();
       router.push(`/plugins${qs ? `?${qs}` : ""}`);
     }, [searchParams, router]);
     ```
   - Compute static counts (after `const allPlugins = Object.values(PLUGINS);`):
     ```
     const totalCount = allPlugins.length;
     const mcpCount = allPlugins.filter((p) => p.type === "mcp").length;
     const pluginCount = allPlugins.filter((p) => p.type === "plugin").length;
     ```
   - Add type filter as FIRST condition in `filtered` computation:
     `if (activeType !== "all" && p.type !== activeType) return false;`
     This goes BEFORE the existing category check. Do NOT reset category when tab changes (locked decision).
   - Pass new props to `<PluginSearch>`:
     ```
     activeType={activeType}
     onTypeChange={handleTypeChange}
     typeCounts={{ total: totalCount, mcp: mcpCount, plugin: pluginCount }}
     ```
